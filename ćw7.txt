CREATE VIEW login_departament AS SELECT hrd.Name, hre.LoginID 
FROM AdventureWorks2017.HumanResources.Department hrd, AdventureWorks2017.HumanResources.EmployeeDepartmentHistory hredh, AdventureWorks2017.HumanResources.Employee hre 
WHERE hre.BusinessEntityID = hredh.BusinessEntityID AND hrd.DepartmentID=hredh.DepartmentID;
